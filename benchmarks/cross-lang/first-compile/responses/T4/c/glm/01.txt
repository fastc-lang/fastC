```c
#include <stdio.h>
#include <stdint.h>
#include <inttypes.h>

int64_t sum_to_n(int32_t n) {
    int64_t sum = 0;
    for (int32_t i = 1; i <= n; i++) {
        sum += i;
    }
    return sum;
}

int main(void) {
    int32_t n = 100000;
    int64_t result = sum_to_n(n);
    printf("%" PRId64 "\n", result);
    return 0;
}
```
