```c
#include <stdio.h>
#include <stdint.h>
#include <inttypes.h>

int64_t sum_to_n(int32_t n) {
    if (n <= 0) return 0;
    /* Use int64_t accumulator to prevent overflow */
    int64_t sum = 0;
    for (int32_t i = 1; i <= n; i++) {
        sum += i;
    }
    return sum;
}

int main(void) {
    int64_t result = sum_to_n(100000);
    printf("%" PRId64 "\n", result);
    return 0;
}
```
